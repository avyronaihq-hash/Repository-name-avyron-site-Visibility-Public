import { Textarea } from '@/components/ui/textarea';

      location: 'Remote / Bangalore',

      department: 'Engineering',
      location: 'Remote / Bangalore',
      type: 'Full-time',
      description: 'Develop and train the AI agents that handle 24/7 sales and support.'
    },
    {
      id: 3,
      title: 'Product Designer',
      department: 'Design',
      location: 'Remote',
      type: 'Full-time',
      description: 'Design intuitive interfaces that simplify complex operational workflows.'
    },
    {
      id: 4,
      title: 'Customer Success Manager',

    if (!formData.name || !formData.email) {

        description: "Please fill in required fields.",

      return;

    toast({
      title: "Application Submitted!",

              <h2 className="text-4xl font-bold text-gray-900 mb-6">

              </h2>
              <p className="text-xl text-gray-600 leading-relaxed">
                We believe in doing great work while maintaining a healthy balance.
              </p>
            </motion.div>

            <div className="grid md:grid-cols-3 gap-8 max-w-6xl mx-auto">
              {benefits.map((benefit, index) => (

                  key={index}
                  initial={{ opacity: 0, y: 20 }}
                  whileInView={{ opacity: 1, y: 0 }}

                    </CardContent>

                </motion.div>

            </div>
          </div>
        </section>

            <motion.div

              viewport={{ once: true }}
              className="max-w-4xl mx-auto text-center mb-16"

              </h2>
            </motion.div>

                  initial={{ opacity: 0, x: -20 }}
                  whileInView={{ opacity: 1, x: 0 }}
                  viewport={{ once: true }}
                  transition={{ delay: index * 0.1 }}
                >
                  <Card className="glass-card hover-lift transition-all">
                    <CardContent className="p-6 sm:p-8 flex flex-col sm:flex-row items-start sm:items-center justify-between gap-6">

                            onChange={(e) => setFormData({...formData, name: e.target.value})}
